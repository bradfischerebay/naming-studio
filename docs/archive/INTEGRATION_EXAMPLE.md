## Option 2: Pre-flight Check (Frontend)

Add name validation before evaluation begins.

### Frontend Component

```typescript
// In your evaluation form component
async function handleSubmit(e: React.FormEvent) {
  e.preventDefault();
  
  setLoading(true);
  
  try {
    // Extract proposed name from brief
    const nameMatch = brief.match(/(?:name|called):\s*["']?([A-Za-z0-9\s]+)["']?/i);
    const proposedName = nameMatch?.[1]?.trim();
    
    if (proposedName) {
      // Pre-flight name check
      const nameCheckResponse = await fetch('/api/check-names', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ names: proposedName })
      });
      
      const nameCheckData = await nameCheckResponse.json();
      const result = nameCheckData.results[0];
      
      if (result.riskLevel === 'critical' || result.riskLevel === 'high') {
        const conflictDetails = result.conflicts
          .map(c => `• ${c.description}`)
          .join('\n');
        
        const proceed = window.confirm(
          `⚠️ Name Conflict Warning\n\n` +
          `The proposed name "${proposedName}" has conflicts:\n\n` +
          `${conflictDetails}\n\n` +
          `Risk Level: ${result.riskLevel.toUpperCase()}\n\n` +
          `Do you want to proceed with evaluation anyway?`
        );
        
        if (!proceed) {
          setLoading(false);
          return;
        }
      }
    }
    
    // Continue with normal evaluation
    const response = await fetch('/api/evaluate', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ brief, model })
    });
    
    const data = await response.json();
    setResult(data);
    
  } catch (error) {
    console.error('Evaluation failed:', error);
  } finally {
    setLoading(false);
  }
}
```

```typescript
// /app/components/NameChecker.tsx
'use client';

import { useState } from 'react';
import type { NameCheckResult } from '@/lib/modules/name-checker';

export function NameChecker() {
  const [name, setName] = useState('');
  const [loading, setLoading] = useState(false);
  const [result, setResult] = useState<NameCheckResult | null>(null);

  async function checkName() {
    if (!name.trim()) return;
    
    setLoading(true);
    try {
      const response = await fetch('/api/check-names', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ names: name })
      });
      
      const data = await response.json();
      setResult(data.results[0]);
    } catch (error) {
      console.error('Name check failed:', error);
    } finally {
      setLoading(false);
    }
  }

  return (
    <div className="name-checker">
      <h2>Check Name Availability</h2>
      
      <div>
        <input
          type="text"
          value={name}
          onChange={(e) => setName(e.target.value)}
          placeholder="Enter proposed product name..."
          onKeyPress={(e) => e.key === 'Enter' && checkName()}
        />
        <button onClick={checkName} disabled={loading}>
          {loading ? 'Checking...' : 'Check Name'}
        </button>
      </div>

      {result && (
        <div className={`result risk-${result.riskLevel}`}>
          <h3>
            {result.isAvailable ? '✅' : '❌'} {result.name}
          </h3>
          
          <div className="risk-badge">
            Risk Level: {result.riskLevel.toUpperCase()}
          </div>

          {result.conflicts.length > 0 && (
            <div className="conflicts">
              <h4>Conflicts Detected:</h4>
              <ul>
                {result.conflicts.map((conflict, i) => (
                  <li key={i} className={`severity-${conflict.severity}`}>
                    <strong>{conflict.type}:</strong> {conflict.description}
                  </li>
                ))}
              </ul>
            </div>
          )}

          <div className="recommendations">
            <h4>Recommendations:</h4>
            <ul>
              {result.recommendations.map((rec, i) => (
                <li key={i}>{rec}</li>
              ))}
            </ul>
          </div>

          {result.ebayData && (
            <div className="ebay-data">
              <h4>eBay Marketplace:</h4>
              <p>{result.ebayData.totalProducts} existing products found</p>
              <a href={result.ebayData.searchUrl} target="_blank" rel="noopener">
                View on eBay →
              </a>
            </div>
          )}

          {result.trademarkData && result.trademarkData.hasConflicts && (
            <div className="trademark-data">
              <h4>Trademark Conflicts:</h4>
              <p>{result.trademarkData.conflicts.length} registered marks found</p>
            </div>
          )}
        </div>
      )}
    </div>
  );
}
```

## Option 4: Backend Integration (Server-Side)

For batch processing or background jobs.

```typescript
// /lib/batch-processor.ts
import { getNameChecker } from '@/lib/modules/name-checker';
import { enhanceG4BatchEvaluation } from '@/lib/modules/g4-integration';

export async function batchProcessNames(candidates: string[]) {
  const nameChecker = getNameChecker();
  
  // Check all names
  const results = await nameChecker.batchCheckNames(candidates);
  
  // Filter to available names
  const available = results.filter(r => r.isAvailable);
  
  // Sort by risk level (lowest first)
  available.sort((a, b) => {
    const levels = { low: 0, medium: 1, high: 2, critical: 3 };
    return levels[a.riskLevel] - levels[b.riskLevel];
  });
  
  return {
    all: results,
    available,
    recommended: available[0], // Best option
  };
}

// Usage in API route or cron job
async function processBatch() {
  const candidates = [
    'AutoSell Pro',
    'Quick Lister',
    'Seller Hub Plus',
    'eBay AutoList' // Should fail
  ];
  
  const results = await batchProcessNames(candidates);
  console.log('Recommended name:', results.recommended?.name);
}
```

## Configuration Options

### 1. Adjust G4 Override Strategy

```typescript
// Conservative: Only fail G4 if critical conflict
if (g4Enhancement.status === 'Fail' && 
    g4Enhancement.nameCheckData?.riskLevel === 'critical') {
  gatekeeperResult.object.G4.status = 'Fail';
}

// Aggressive: Fail on any high risk
if (g4Enhancement.nameCheckData?.riskLevel === 'high' ||
    g4Enhancement.nameCheckData?.riskLevel === 'critical') {
  gatekeeperResult.object.G4.status = 'Fail';
}

// Augment only: Never override LLM decision
gatekeeperResult.object.G4.reasoning += 
  `\n\n${g4Enhancement.reasoning}`;
```
